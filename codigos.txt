let timerVal = 20;
let dashboardAtivo = false;

// Configurações Globais dos Gráficos
const cfg = { responsive: true, maintainAspectRatio: false, animation: { duration: 800 } };

let chartSaude = new Chart(document.getElementById('chartSaude'), {
    type: 'doughnut',
    data: { labels: ['OK', 'Alarme'], datasets: [{ data: [{{ ok }}, {{ alarme }}], backgroundColor: ['#00ff88', '#ff4444'], borderWidth: 0 }] },
    options: cfg
});

let chartSetores = new Chart(document.getElementById('chartSetores'), {
    type: 'bar',
    data: { labels: {{ alarmes_por_setor.keys()|list|tojson }}, datasets: [{ label: 'Alarmes', data: {{ alarmes_por_setor.values()|list|tojson }}, backgroundColor: '#3b82f6' }] },
    options: cfg
});

let chartItens = new Chart(document.getElementById('chartItens'), {
    type: 'bar',
    data: { 
        labels: {{ variaveis_alarme.keys()|list|tojson }}, 
        datasets: [{ label: 'Qtd Falhas', data: {{ variaveis_alarme.values()|list|tojson }}, backgroundColor: '#f59e0b' }] 
    },
    options: { 
        ...cfg, 
        scales: { 
            y: { beginAtZero: true, ticks: { precision: 0 } },
            x: { ticks: { autoSkip: false, maxRotation: 45, minRotation: 45 } }
        } 
    }
});

function iniciarDashboard() {
    document.getElementById('camadaAtivacao').style.display = 'none';
    dashboardAtivo = true;
    atualizarDados();
}

async function atualizarDados() {
    if(!dashboardAtivo) return;
    try {
        const response = await fetch('/api_dashboard?t=' + Date.now());
        const d = await response.json();

        // 1. Atualiza KPIs
        document.getElementById('val-total').innerText = d.total;
        document.getElementById('val-ok').innerText = d.ok;
        document.getElementById('val-alarme').innerText = d.alarme;

        // 2. Atualiza Gráfico Saúde
        chartSaude.data.datasets[0].data = [d.ok, d.alarme];
        chartSaude.update();

        // 3. Atualiza Gráfico Setores
        chartSetores.data.labels = Object.keys(d.alarmes_por_setor);
        chartSetores.data.datasets[0].data = Object.values(d.alarmes_por_setor);
        chartSetores.update();

        // 4. ATUALIZAÇÃO DO GRÁFICO DE ITENS (TOP FALHAS) - CORREÇÃO DEFINITIVA
        chartItens.data.labels = Object.keys(d.variaveis_alarme);
        chartItens.data.datasets[0].data = Object.values(d.variaveis_alarme);
        chartItens.update();

        // 5. Atualiza Tabela
        let htmlTabela = "";
        d.lista.forEach(reg => {
            htmlTabela += `<tr>
                <td style="width:25%"><b>${reg.Equipamento}</b></td>
                <td style="width:25%">${reg.Item}</td>
                <td style="color:red; font-weight:800; width:10%">${reg.Valor}</td>
                <td style="width:20%">${reg.Setor}</td>
                <td style="width:20%">${reg.Data}</td>
            </tr>`;
        });
        // Duplica para efeito de scroll infinito
        document.getElementById('wrapperTabela').innerHTML = `<table><tbody>${htmlTabela}${htmlTabela}</tbody></table>`;

    } catch (e) { console.error("Erro na sincronização:", e); }
}

setInterval(() => {
    document.getElementById('clock').innerText = new Date().toLocaleTimeString('pt-BR');
    if(dashboardAtivo) {
        timerVal--;
        if(timerVal <= 0) { timerVal = 20; atualizarDados(); }
        document.getElementById('timer').innerText = timerVal;
    }
}, 1000);
</script>
